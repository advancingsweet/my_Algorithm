2. 利用偏移量走 “十字”、“九宫格”、“日字”

   ```C++
   //遍历九宫格
   for(int i=-1;i<=1;i++)
       for(int j=-1;j<=1;j++)	
             int dx = x+i,dy = y+j;
   
   
   //马走日
   int dx[8] = {-1,-1,-2,-2,1,1,2,2};
   int dy[8] = {2,-2,1,-1,2,-2,1,-1};
   for(int i=0;i<8;i++)
       int x1 = x + dx[i], y1 = y + dy[i];
   
   //"十字"格
   int dx[5] = {-1, 0, 1, 0, 0}, dy[5] = {0, 1, 0, -1, 0};
   for(int i=0;i<5;i++)
       int x1 = x + dx[i], y1 = y + dy[i];
   ```
